```sql
  -- Participants
  user_1_id UUID REFERENCES users(id),
  user_2_id UUID REFERENCES users(id), -- NULL for solo mode
  
  -- Session details
  duration INTEGER NOT NULL, -- Minutes (15, 25, 50, 90)
  mode TEXT NOT NULL, -- 'duo', 'solo', 'lounge'
  theme_id INTEGER, -- References themes
  music_playlist TEXT,
  
  -- Status
  status TEXT DEFAULT 'matching', -- matching, active, completed, cancelled
  started_at TIMESTAMP,
  ended_at TIMESTAMP,
  
  -- Goals
  user_1_goal TEXT,
  user_2_goal TEXT,
  
  -- Ratings (1-5 stars)
  user_1_rating INTEGER,
  user_2_rating INTEGER,
  
  -- Completion
  user_1_completed BOOLEAN DEFAULT false,
  user_2_completed BOOLEAN DEFAULT false,
  
  created_at TIMESTAMP DEFAULT NOW()
);

-- Matching Queue Table (Realtime)
CREATE TABLE matching_queue (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES users(id),
  
  duration INTEGER, -- Preferred session length
  theme_preference INTEGER,
  mode TEXT DEFAULT 'duo', -- duo, lounge
  
  priority INTEGER DEFAULT 0, -- Higher = matched faster (no-show victims)
  
  created_at TIMESTAMP DEFAULT NOW()
);

-- Achievements Table
CREATE TABLE achievements (
  id INTEGER PRIMARY KEY,
  name TEXT NOT NULL,
  description TEXT,
  icon TEXT, -- Emoji or image URL
  requirement JSONB, -- { "type": "streak", "value": 7 }

CREATE TABLE user_achievements (
  user_id UUID REFERENCES users(id),
```

### **Matching Algorithm**

```javascript
// Supabase Edge Function: /functions/match-users

async function matchUsers(userId, preferences) {
  const { duration, theme, mode } = preferences;
  
  // 1. Add user to queue
  await supabase.from('matching_queue').insert({
    user_id: userId,
    duration,
    theme_preference: theme,
    mode,
    priority: await getPriority(userId) // No-show victims get +10
  });
  
  // 2. Find compatible partner
  const match = await supabase
    .from('matching_queue')
    .select('*')
    .neq('user_id', userId) // Not yourself
    .eq('duration', duration) // Same duration
    .eq('mode', mode) // Same mode
    .gte('created_at', new Date(Date.now() - 30000)) // Last 30s
    .order('priority', { ascending: false }) // Priority first
    .order('created_at', { ascending: true }) // FIFO
    .limit(1)
    .single();
  
  if (!match) {
    // No match found, wait or offer solo mode
    setTimeout(() => checkAgainOrOfferSolo(userId), 15000);
    return null;
  }
  
  // 3. Create session
  const session = await supabase.from('sessions').insert({
    user_1_id: userId,
    user_2_id: match.user_id,
    duration,
    mode,
    theme_id: theme,
    status: 'matched'
  }).select().single();
  
  // 4. Remove both from queue
  await supabase.from('matching_queue').delete()
    .in('user_id', [userId, match.user_id]);
  
  // 5. Notify both users (Realtime)
  await notifyMatch(userId, match.user_id, session.id);
  
  return session;
}
```
